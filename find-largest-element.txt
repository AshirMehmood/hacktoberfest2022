#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int largest(int* a,int* b, int* c,int* d){
    int arr[4]={*a,*b,*c,*d};
    int max=0;
    for(int i=0;i<=4;i++){
        if(arr[i]>max){
            max=arr[i];
        }
    }
    cout<<max;
    return max;
}

int main() {
   int a,b,c,d;
   cin>>a>>b>>c>>d;
   largest(&a,&b,&c,&d);
    return 0;
}
